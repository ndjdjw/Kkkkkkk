async function searchVideos() { const query = document.getElementById('searchInput').value.trim().toLowerCase(); const resultsDiv = document.getElementById('results'); const searchBtn = document.getElementById('searchBtn'); if (!query) { alert('الرجاء إدخال كلمة البحث'); return; } searchBtn.disabled = true; searchBtn.textContent = 'جاري البحث...'; try { // تحميل الفيديوهات إذا لم تكن محملة if (!window.videos) { const response = await fetch('/data/videos.json'); const data = await response.json(); window.videos = data.videos; } // فلترة النتائج const filtered = window.videos.filter(v => v.title.toLowerCase().includes(query) ); if (filtered.length === 0) { resultsDiv.innerHTML = '

لا توجد نتائج

'; return; } // عرض النتائج let html = ''; filtered.forEach(v => { html += `
${v.title}
${v.channel}
مشاهدة
`; }); resultsDiv.innerHTML = html; } catch (error) { console.error('خطأ:', error); resultsDiv.innerHTML = '

حدث خطأ في تحميل الفيديوهات

'; } searchBtn.disabled = false; searchBtn.textContent = 'بحث'; }
