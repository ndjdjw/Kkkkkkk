async function searchVideos() { const query = document.getElementById('searchInput').value.trim(); if (!query) { alert('الرجاء إدخال كلمة البحث'); return; } // إظهار شاشة التحميل... searchBtn.disabled = true; searchBtn.textContent = 'جاري البحث...'; resultsGrid.innerHTML = `

جاري البحث...

`; try { // اقرأ من الملف المحلي const response = await fetch('/data/videos.json'); const data = await response.json(); // فلترة النتائج حسب عنوان الفيديو const filteredVideos = data.videos.filter(video => video.title.toLowerCase().includes(query.toLowerCase()) ); // إذا ما في نتائج if (filteredVideos.length === 0) { resultsGrid.innerHTML = `

لا توجد نتائج لـ "${query}"

`; return; } // عرض النتائج displayResults({ items: filteredVideos }); } catch (error) { console.error('خطأ في جلب الفيديوهات:', error); resultsGrid.innerHTML = `

حدث خطأ في تحميل الفيديوهات.

`; } finally { searchBtn.disabled = false; searchBtn.textContent = 'بحث'; } }
